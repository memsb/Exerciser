	 * @return String the location uri

	 * Gets the mime type
	 * @param View the view to combine the type with
	 * @return String the mime type

	 * @param View the view
	 * @return String the generated document

	public function generateDocument($view){
		if( ! $view )
			return '';
		$view->startDocument();
		$this->addProperties($view);
