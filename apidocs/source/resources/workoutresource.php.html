require_once dirname(__FILE__) . '/../classes/workout.php';
require_once dirname(__FILE__) . '/../classes/user.php';
require_once dirname(__FILE__) . '/../lib/MultiViewResource.php';

/**
 * @namespace Tonic\Exerciser\Workouts\Workout
 * @uri /workouts/([0-9]+)
 */
class WorkoutResource extends MultiViewResource {

	/**
	*
	*
	*/
	public function __construct($parameters){
		parent::__construct($parameters);
		$this->model = new Workout($this->db);
	}
    
	/**
	* Sets the model to be controlled
	* @param Activity model
	*/
	public function setModel($model){
		$this->model = $model;
	}  

	/**
	* Handle a GET request for this resource
	* @param Request request
	* @return Response
	*/
	public function get($request) {
		$response = new Response($request);
		$ID = $this->parameters[0];
		$this->model->setLocation($request->uri);
		$this->view = $this->get_view($request);
		try{
			$this->model->load($ID);
			$response = new Response($request);
			$response->code = Response::OK;
			$response->addHeader('Content-type', $this->model->type($this->view));
			$response->body = $this->model->generateDocument($this->view);  
		}catch (Exception $e) {
			$response->code = Response::NOTFOUND;
		}
		return $response;
	}

	/**
	*
	*
	*/
	public function post($request){
		$response = new Response($request);
		$ID = $this->parameters[0];

		try{
			$user = new User('');
			$user->load($this->model->user_id);
			$this->isSecured($user->user_id, $user->password);

			$this->model->parse($request->data, $this->view);
			$this->model->create();		

			$response->code = Response::CREATED;
			$response->addHeader('Location', $this->model->location());
		}catch (Exception $e) {
			$response->code = Response::UNSUPPORTEDMEDIATYPE;
		}
		return $response;
	}

	/**
	* Handle a PUT request for this resource
	* @param Request request
	* @return Response
	*/
	public function put($request) {
		$response = new Response($request);
		$ID = $this->parameters[0];	

		try{
			$this->model->load($ID);
			$user = new User('');
			$user->load($this->model->user_id);
			$this->isSecured($user->user_id, $user->password);

			$this->model->parse($request->data, $this->view);
			$this->model->update();		
			
			$response->code = Response::OK;
			$response->addHeader('Location', $this->model->location());
		}catch (UnexpectedValueException $e) {
			$response->code = Response::UNSUPPORTEDMEDIATYPE;
		}catch (Exception $e) {
			$response->code = Response::NOTFOUND;
		}
		return $response;
	}

	/**
	* Handle a DELETE request for this resource
	* @param Request request
	* @return Response
	*/
	public function delete($request) {
		$response = new Response($request);
		$ID = $this->parameters[0];

		try{
			$this->model->load($ID);
			$user = new User('');
			$user->load($this->model->user_id);
			$this->isSecured($user->user_id, $user->password);

			$this->model->delete();
			$response->code = Response::NOCONTENT;
		}catch (Exception $e) {
			$response->code = Response::NOTFOUND;
		}
		return $response;
	}
    
}
