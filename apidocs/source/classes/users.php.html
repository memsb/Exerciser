require_once dirname(__FILE__) . '/../lib/CRUD.php';
require_once dirname(__FILE__) . '/user.php';

/**
*
*
*/
class Users extends CRUD {

	private $type = 'users';

	protected $users = array();

	/**
	*
	*
	*/
	public function __construct($db){
		$this->db = $db;
		if( $this->db == NULL )
			throw new BadMethodCallException('Database not specified.');
		$this->db->connect();
	}

	/**
	*
	*
	*/
	public function setLocation($uri){
		$this->uri = $uri;
	}

	/**
	*
	*
	*/
	public function size(){
		return count($this->users);
	}

	/**
	*
	*
	*/
	public function addUser(User $user){
		$this->users[] = $user;
	}

	/**
	*
	*
	*/
	public function load(){
		$result = $this->db->query("SELECT User_ID FROM Users");		
		foreach($result as $row){
			$id = $row['User_ID'];
			$user = new User($this->db);
			$user->setLocation($this->uri . '/' . $id);
			$user->load($id);
			$this->users[] = $user;
		}
	}

	/**
	*
	*
	*/
	public function location(){
		return $this->uri;
	}
	
	/**
	*
	*
	*/
	public function generateDocument($view){
		$view->startDocument();
		$view->startElement('users');
		$i = 0;
		foreach($this->users as $user){
			$view->startElement($i++);		
			$user->addProperties($view);
			$view->endElement();
		}

		$view->endElement();
		return $view->toString();
	}

	/**
	*
	*
	*/
	public function type($view){
		return $view->type() . $this->type;
	}
}

?>
